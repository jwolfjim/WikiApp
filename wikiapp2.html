<script>


$(document).ready(function() {

  //Creates variable that stores input on click

  // Get the input field

var input = document.getElementById("inputText");


// Execute a function when the user releases a key on the keyboard

input.addEventListener("keyup", function(event) {

  // Cancel the default action, if needed

  event.preventDefault();

  // Number 13 is the "Enter" key on the keyboard

  if (event.keyCode === 13) {

    // Trigger the button element with a click

    document.getElementById("search").click();

  }

});

  $('#search').click(function(){

    var wikiSearch = $('#inputText').val();

    //console.log(wikiSearch);

    var searchURL = "https://en.wikipedia.org/w/api.php?action=opensearch&search=" + wikiSearch +"&format=json&callback=?";

  $.ajax({

    type: "GET",

    url: searchURL,

    contentType: "application/json; charset=utf-8",

    dataType: "json",

    success: function(data, textStatus, jqXHR){

      console.log(data);

      var res1 = data[1][0]; //This gets the name of the first hit

      var resText1 = data[2][0]; //This gets a short description of the first hit

      var res2 = data[1][1]; //This gets the name of the first hit

      var resText2 = data[2][1]; //This gets a short description of the first hit

      var res3 = data[1][2]; //This gets the name of the first hit

      var resText3 = data[2][2]; //This gets a short description of the first hit

      $('#resName1').html(res1);

      $('#resText1').html(resText1); 

      $('#resName2').html(res2);

      $('#resText2').html(resText2); 

      $('#resName3').html(res3);

      $('#resText3').html(resText3); 

      $('#resName1').attr("href", data[3][0]);

      $('#resName2').attr("href", data[3][1]);

      $('#resName3').attr("href", data[3][2]);


      var res4 = data[1][3]; //This gets the name of the first hit

      var resText4 = data[2][3]; //This gets a short description of the first hit

      var res5 = data[1][4]; //This gets the name of the first hit

      var resText5 = data[2][4]; //This gets a short description of the first hit

      var res6 = data[1][5]; //This gets the name of the first hit

      var resText6 = data[2][5]; //This gets a short description of the first hit

      $('#resName4').html(res4);

      $('#resText4').html(resText4); 

      $('#resName5').html(res5);

      $('#resText5').html(resText5); 

      $('#resName6').html(res6);

      $('#resText6').html(resText6); 

      $('#resName4').attr("href", data[3][3]);

      $('#resName5').attr("href", data[3][4]);

      $('#resName6').attr("href", data[3][5]);


          }    

    }

    

    )

  })

  

  

});
